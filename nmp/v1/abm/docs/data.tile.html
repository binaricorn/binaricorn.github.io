            function urlToTileCoords(url) {
                var coordRegex = /.*\/(\d+)\/(\d+)\/(\d+)\.png$/;
                var coords = coordRegex.exec(url);
                return coords && {
                    z: coords[1],
                    x: coords[2],
                    y: coords[3]
                };
            }
